Installation & Usage

1. virtualenv -p $(which python3) venv
2. sudo chmod 777 ./venv/bin/activate
3. source ./venv/bin/activate
4. ./venv/bin/pip install -r requirements.txt
5. ./venv/bin/python3 spider.py --link=http://delos.aueb.gr/opendelos/search\?crs\=bfd90397 --pages=1
